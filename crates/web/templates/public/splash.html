{# Rust #} 
use std::net::TcpListener;
use std::io::Write;

fn main() {
    let listener = TcpListener::bind("0.0.0.0:8080").unwrap();

    for stream in listener.incoming() {
        let mut stream = stream.unwrap();
        let response = b"HTTP/1.1 200 OK\r\n\r\nhello from liquid metal";
        stream.write_all(response).ok();
    }
}
{# Zig #} 
const std = @import("std");

pub fn main() !void {
    const addr = try std.net.Address.resolveIp("0.0.0.0", 8080);
    var server = try addr.listen(.{});
    defer server.deinit();

    while (true) {
        const conn = try server.accept();
        _ = conn.stream.write("HTTP/1.1 200 OK\r\n\r\nhello from liquid metal");
        conn.stream.close();
    }
}

{# ── WHY ─────────────────────────────────────────────────────────────── #}

// the problem

Deploy platforms forgot about compiled languages.

Most platforms want a Dockerfile, a runtime, or a framework. If you write Go, Rust, or Zig, you already have a binary. You shouldn't need to wrap it in a container just to deploy it. Liquid Metal takes your compiled binary and runs it — directly on hardware or as WebAssembly.

Metal — Firecracker

Your binary gets its own Firecracker microVM. Dedicated vCPU, dedicated RAM, its own kernel. Boots in 125ms. No shared tenants.

Liquid — Wasm

Compile to .wasm and run it in Wasmtime. Sub-millisecond cold starts, per-invocation billing. No VM overhead for lightweight workloads.

Your binary, as-is

No SDK. No wrapper. No runtime dependency. If go build, cargo build, or zig build produces it, we run it.

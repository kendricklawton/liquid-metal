{# Rust #} 
use std::net::TcpListener;
use std::io::Write;

fn main() {
    let listener = TcpListener::bind("0.0.0.0:8080").unwrap();

    for stream in listener.incoming() {
        let mut stream = stream.unwrap();
        let body = "hello from liquid metal";
        let response = format!(
            "HTTP/1.1 200 OK\r\n\r\n{body}"
        );
        stream.write_all(response.as_bytes()).ok();
    }
}
{# Zig #} 
const std = @import("std");

pub fn main() !void {
    const addr = try std.net.Address.resolveIp("0.0.0.0", 8080);
    var server = try addr.listen(.{});
    defer server.deinit();

    while (true) {
        const conn = try server.accept();
        const resp =
            "HTTP/1.1 200 OK\r\n\r\nhello from liquid metal";
        _ = conn.stream.write(resp);
        conn.stream.close();
    }
}

{# ── WHY ─────────────────────────────────────────────────────────────── #}

// the problem

Deploy platforms forgot about compiled languages.

Most platforms want a Dockerfile, a runtime, or a framework. If you write Go, Rust, or Zig, you already have a binary. You shouldn't need to wrap it in a container just to deploy it. Liquid Metal takes your compiled binary and runs it — on a dedicated VM or as serverless Wasm.

Metal — Dedicated VM

Your binary runs in a KVM-isolated VM on bare metal hardware. Always on, no cold starts, fixed monthly price. Real Linux kernel, real networking.

Liquid — Serverless Wasm

Compile to .wasm and run it in Wasmtime. Sub-millisecond cold starts, scale to zero, per-invocation billing. For lightweight stateless workloads.

Your binary, as-is

No SDK. No wrapper. No runtime dependency. If go build, cargo build, or zig build produces it, Liquid Metal runs it.

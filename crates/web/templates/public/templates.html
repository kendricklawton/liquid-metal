// metal — dedicated vms

Metal runs your static binary in a Firecracker microVM. Compile to x86_64-unknown-linux-musl, listen on a port, and deploy. The VM stays running — no cold starts.

{# Go HTTP server #}

main.go

package main

import (
    "fmt"
    "net/http"
    "os"
)

func main() {
    port := os.Getenv("PORT")
    if port == "" {
        port = "8080"
    }
    http.HandleFunc("/", func(w http.ResponseWriter, r *http.Request) {
        fmt.Fprintln(w, "hello from liquid metal")
    })
    http.ListenAndServe(":"+port, nil)
}

liquid-metal.toml

[service]
name = "go-hello"
engine = "metal"
port = 8080

[build]
command = "CGO_ENABLED=0 GOOS=linux GOARCH=amd64 go build -o app ."
output = "app"

Deploy

flux init --engine metal --name go-hello
flux deploy
{# Rust HTTP server #}

src/main.rs

use std::net::TcpListener;
use std::io::Write;

fn main() {
    let port = std::env::var("PORT")
        .unwrap_or_else(|_| "8080".into());
    let addr = format!("0.0.0.0:{}", port);
    let listener = TcpListener::bind(&addr).unwrap();

    for stream in listener.incoming() {
        let mut s = stream.unwrap();
        s.write_all(
            b"HTTP/1.1 200 OK\r\n\r\nhello from liquid metal"
        ).ok();
    }
}

liquid-metal.toml

[service]
name = "rust-hello"
engine = "metal"
port = 8080

[build]
command = "cargo build --target x86_64-unknown-linux-musl --release"
output = "target/x86_64-unknown-linux-musl/release/rust-hello"

Deploy

flux init --engine metal --name rust-hello
flux deploy

// liquid — serverless wasm

Liquid runs your .wasm module in Wasmtime. Each request spawns a fresh instance. Read the request from stdin and env vars (WAGI/CGI style), write the response to stdout. Scale to zero when idle, wake in under 10ms.

{# Go WAGI handler #}

main.go

package main

import (
    "fmt"
    "os"
)

func main() {
    // WAGI: read request from env vars,
    // write CGI response to stdout.
    method := os.Getenv("REQUEST_METHOD")
    path := os.Getenv("PATH_INFO")

    fmt.Println("Content-Type: application/json")
    fmt.Println()
    fmt.Printf("{\"method\":\"%s\",\"path\":\"%s\"}\n", method, path)
}

liquid-metal.toml

[service]
name = "go-wagi"
engine = "liquid"

[build]
command = "GOOS=wasip1 GOARCH=wasm go build -o main.wasm ."
output = "main.wasm"

Deploy

flux init --engine liquid --name go-wagi
flux deploy
{# Rust WAGI handler #}

src/main.rs

use std::env;

fn main() {
    let method = env::var("REQUEST_METHOD")
        .unwrap_or_default();
    let path = env::var("PATH_INFO")
        .unwrap_or_default();

    // CGI response: headers, blank line, body
    println!("Content-Type: application/json");
    println!();
    println!(
        r#"{"method":"{}","path":"{}"}"#,
        method, path
    );
}

liquid-metal.toml

[service]
name = "rust-wagi"
engine = "liquid"

[build]
command = "cargo build --target wasm32-wasip1 --release"
output = "target/wasm32-wasip1/release/rust-wagi.wasm"

Deploy

flux init --engine liquid --name rust-wagi
flux deploy

// how liquid (wagi) works

The WAGI Protocol

Liquid uses WAGI (WebAssembly Gateway Interface) — the same idea as CGI, but for Wasm. Each HTTP request spawns a fresh Wasm instance. The request is passed via environment variables and stdin. The response is read from stdout. No HTTP library needed.

Request → Env Vars

REQUEST_METHOD GET, POST, PUT, DELETE, etc.
PATH_INFO Request path (e.g. /users/123)
QUERY_STRING Query parameters (e.g. page=2&limit=10)
CONTENT_TYPE Request content type header
CONTENT_LENGTH Request body size in bytes
HTTP_* All HTTP headers (e.g. HTTP_AUTHORIZATION)
stdin Request body (for POST/PUT)

Response → stdout

# Write headers, blank line, then body to stdout:
Content-Type: application/json
Status: 200 OK

{"message": "hello"}
